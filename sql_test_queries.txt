1. SELECT DISTINCT inspectors.name, inspectors.phone FROM inspectors 
	JOIN inspections ON inspectors.id = inspections.inspector 
	JOIN inspection_parcel ON inspections.id = inspection_parcel.inspection_id
	JOIN parcels ON inspection_parcel.parcel_id = parcels.id
	WHERE parcels.ppn = '812-12-027';

Result: 
Name: Mia Brown Phone: (216) 555-4321
Name: Emma Johnson Phone: (216) 555-1234

2. SELECT SUM(inspections.cost_per_parcel) AS total_cost FROM inspections
	JOIN inspection_parcel ON inspections.id = inspection_parcel.inspection_id
	JOIN parcels on inspection_parcel.parcel_id = parcels.id
	WHERE parcels.address LIKE '%East Cleveland, OH%';

Result: 20900.00

3. SELECT DISTINCT parcels.address FROM parcels
	LEFT JOIN inspection_parcel ON inspection_parcel.parcel_id = parcels.id
	LEFT JOIN inspections ON inspection_parcel.inspection_id = inspections.id
	WHERE inspections.date IS NULL OR inspections.date < '2023-01-01';
